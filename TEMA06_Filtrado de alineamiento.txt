# Solicitamos un nodo 
srun --partition=cpu --nodes=1 --ntasks=1 --cpus-per-task=1 --mem=2G --pty bash -i
# Cargamos conda
module load conda
# Y activamos el ambiente
source activate gatk_env

----------------------------------------------------------------------------------------
### ¡¡¡NO CORRER!!! ###
# Realizaremos el filtrado utilizando la herramienta de samtools view
samtools view \
 --threads 4 \
 -b -h  -q 20 \
 -F UNMAP,MUNMAP,SECONDARY,QCFAIL,DUP,SUPPLEMENTARY \
 -T ../ref/reference.fasta \
 -o filtered_SRR13166685.bam \
 aligned_SRR13166685.sam
# Después se tienen que ordenar
samtools sort \
 --threads 4 \
 --reference ../ref/reference.fasta \
 -o sorted_SRR13166685.bam \
 filtered_SRR13166685.bam
# E indexar con samtools index (la opcion -@ es el numero de threads)
samtools index sorted_SRR13166685.bam
# Despues realizaremos el filtrado de lecturas duplicadas
gatk MarkDuplicates \
 -I sorted_SRR13166685.bam \
 -O dedup_SRR13166685.bam \
 -M metrics_SRR13166685.txt \
 --REMOVE_DUPLICATES true
# Y se indexa nuevamente con samtools index
samtools index dedup_SRR13166685.bam
-------------------------------------------------------------------------------------------
