 SRR13166685_subChr6_1.fastq SRR13166685_subChr6_2.fastq \
 clean_SRR13166685_subChr6_1.fastq unpair_SRR13166685_subChr6_1.fastq \
 clean_SRR13166685_subChr6_2.fastq unpair_SRR13166685_subChr6_2.fastq \

# Hay que evaluar la calidad de los archivos de salida
fastqc clean_SRR131666*_subChr6_1.fastq
